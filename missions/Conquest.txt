conquer_duskwood = {
	type = country

	category = MIL
	ai_mission = yes
	
	allow = {
        # Only allow warmonger races to use these
        is_savage = yes
		region_duskwood = {
			NOT = { owned_by = ROOT }
            # Only allow if the cultures are sufficiently different
            OR = {
                NOT = { culture_group = ROOT }
                NOT = { religion_group = ROOT }
            }
		}
	}
	abort = {
		always = no
	}
	success = {
		region_duskwood = {
			type = all
			owned_by = ROOT
		}
	}
	chance = {
		factor = 1000
        
        modifier = {
			factor = 10
			OR = {
                tag = A41
                tag = A40
                tag = A43
            }
		}
        
        # Disallow if they own no provinces on the continent
		modifier = {
			factor = 0
            
            NOT = {
                any_owned_province = {
                    superregion = eastern_kingdom_superregion
                }
            }
		}
        
        modifier = {
			factor = 0
            
            NOT = {
                any_owned_province = {
                    province_group = azeroth_provincegroup
                }
            }
		}

	}
	immediate = {
		region_duskwood = {
			add_claim = ROOT
		}
	}
	abort_effect = {
		region_duskwood = {
			remove_claim = ROOT
		}
	}
	effect = {
		add_adm_power = 100
        add_dip_power = 100
        add_mil_power = 100
	}
}
