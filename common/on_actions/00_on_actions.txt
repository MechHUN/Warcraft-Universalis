# country
on_startup = {
  events = {
    meta.1  # Discovery
  }
  if = {
    limit = {
      is_year = 591
        NOT = {
          has_global_flag = the_dark_portal_has_started
        }
    }
    set_global_flag = the_dark_portal_has_started
  }
  if = {
    limit = {
      has_country_modifier = alliance_member
    }
    events = { faction_changes.1 }
  }
  if = {
    limit = {
      has_country_modifier = horde_member
    }
    events = { faction_changes.2 }
  }
  if = {
    limit = {
      has_country_modifier = neutral_member
    }
    events = { faction_changes.3 }
  }
}
